name: Generate Daily Key

on:
  schedule:
    - cron: '0 0 * * *'
  workflow_dispatch:

permissions:
  contents: write

jobs:
  generate-key:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Generate unique key
        run: |
          touch Key
          touch used_keys.txt

          while true; do
            RANDOM_STR=$(head /dev/urandom | tr -dc A-Z0-9 | head -c 12)
            NEW_KEY="MOONCONHUB-${RANDOM_STR}"

            if ! grep -qx "$NEW_KEY" used_keys.txt; then
              break
            fi
          done

          echo "$NEW_KEY" > Key
          echo "$NEW_KEY" >> used_keys.txt

          echo "สร้างคีย์ใหม่: $NEW_KEY"

      - name: Push key
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"

          git add Key used_keys.txt
          git commit -m "Update daily key" || echo "No changes"
          git push
